# 04 — 数据库设计

> 推荐使用 **PostgreSQL**（生产）/ **SQLite**（本地开发）。  
> ORM 推荐：Prisma（Node.js）或 SQLAlchemy（Python）。

## 表结构

### `devices` — 设备注册表

```sql
CREATE TABLE devices (
  id            VARCHAR(32) PRIMARY KEY,          -- 如 dev_abc123xyz
  device_name   VARCHAR(128) NOT NULL,            -- 如 Device-Win-1094
  hostname      VARCHAR(255),
  platform      VARCHAR(32) NOT NULL DEFAULT 'win32',
  mac_hash      VARCHAR(64),                      -- SHA256(MAC地址)，用于防重复注册
  client_version VARCHAR(32),                     -- 注册时客户端版本
  created_at    TIMESTAMP NOT NULL DEFAULT NOW(),
  last_seen_at  TIMESTAMP,                        -- 最近一次上报时间
  is_active     BOOLEAN NOT NULL DEFAULT TRUE,
  notes         TEXT
);

CREATE INDEX idx_devices_mac_hash ON devices(mac_hash);
CREATE INDEX idx_devices_last_seen ON devices(last_seen_at);
```

---

### `api_keys` — API Key 管理

```sql
CREATE TABLE api_keys (
  id          VARCHAR(32) PRIMARY KEY,            -- 如 key_abc123
  device_id   VARCHAR(32) NOT NULL REFERENCES devices(id) ON DELETE CASCADE,
  key_hash    VARCHAR(64) NOT NULL UNIQUE,        -- SHA256(api_key明文)，不存明文
  key_prefix  VARCHAR(16) NOT NULL,               -- 如 nk_stat_xxxx（前8位，用于日志展示）
  created_at  TIMESTAMP NOT NULL DEFAULT NOW(),
  expires_at  TIMESTAMP,                          -- NULL = 永不过期
  revoked_at  TIMESTAMP,
  last_used_at TIMESTAMP,
  scope       VARCHAR(128) DEFAULT 'read,write'  -- 权限范围（预留）
);

CREATE INDEX idx_api_keys_device_id ON api_keys(device_id);
CREATE INDEX idx_api_keys_key_hash ON api_keys(key_hash);
```

> **安全要求**：永远不要将 api_key 明文存入数据库，只存其 SHA256 哈希。服务端验证时，对请求头中的 key 做哈希后与表中 key_hash 比对。

### `status_reports` — 设备状态上报记录

```sql
CREATE TABLE status_reports (
  id              BIGSERIAL PRIMARY KEY,
  device_id       VARCHAR(32) NOT NULL REFERENCES devices(id) ON DELETE CASCADE,
  reported_at     TIMESTAMP NOT NULL,              -- 客户端上报时间
  received_at     TIMESTAMP NOT NULL DEFAULT NOW(),
  cpu_usage       REAL,                            -- 0~100
  cpu_temp        REAL,                            -- 摄氏度
  memory_usage    REAL,                            -- 0~100
  memory_used_mb  INT,
  memory_total_mb INT,
  battery_level   SMALLINT,                        -- 0~100，-1 表示无电池
  is_charging     BOOLEAN,
  active_app      VARCHAR(255),
  active_process  VARCHAR(255),
  upload_kbps     INT,
  download_kbps   INT,
  latency_ms      INT,
  storage_used_gb INT,
  storage_total_gb INT,
  raw_payload     JSONB                            -- 原始上报 JSON，防字段遗漏
);

-- 分区建议（数据量大时按月分区）
CREATE INDEX idx_status_device_time ON status_reports(device_id, reported_at DESC);

-- 数据保留策略：建议只保留最近 90 天，可通过定时任务清理
-- DELETE FROM status_reports WHERE received_at < NOW() - INTERVAL '90 days';
```

### `screenshots` — 截图记录

```sql
CREATE TABLE screenshots (
  id              VARCHAR(40) PRIMARY KEY,         -- 如 ss_20260308_104215_001
  device_id       VARCHAR(32) NOT NULL REFERENCES devices(id) ON DELETE CASCADE,
  captured_at     TIMESTAMP NOT NULL,
  uploaded_at     TIMESTAMP NOT NULL DEFAULT NOW(),
  file_path       TEXT NOT NULL,                   -- 服务器相对存储路径
  url             TEXT NOT NULL,                   -- 对外访问 URL
  thumbnail_url   TEXT,
  resolution      VARCHAR(16),                     -- 如 1920x1080
  file_size_bytes INT,
  active_app      VARCHAR(255),
  monitor_index   SMALLINT DEFAULT 1,
  is_deleted      BOOLEAN NOT NULL DEFAULT FALSE,
  deleted_at      TIMESTAMP
);

CREATE INDEX idx_screenshots_device_time ON screenshots(device_id, captured_at DESC);
```

### `update_channels` — 更新通道配置

```sql
CREATE TABLE update_channels (
  id          VARCHAR(16) PRIMARY KEY,   -- 'stable' | 'beta' | 'nightly'
  display_name VARCHAR(32) NOT NULL,
  description TEXT,
  github_repo VARCHAR(256),             -- 如 your-org/neko-status
  tag_pattern VARCHAR(128),             -- 如 ^v\d+\.\d+\.\d+$
  is_prerelease BOOLEAN NOT NULL DEFAULT FALSE,
  is_active   BOOLEAN NOT NULL DEFAULT TRUE,
  created_at  TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at  TIMESTAMP NOT NULL DEFAULT NOW()
);

-- 初始数据
INSERT INTO update_channels VALUES
  ('stable',  '稳定版', '经过完整测试的正式发布版本', 'your-org/neko-status', '^v\d+\.\d+\.\d+$', FALSE, TRUE, NOW(), NOW()),
  ('beta',    'Beta 版', '新功能预览，可能存在已知问题', 'your-org/neko-status', '^v\d+\.\d+\.\d+-beta', TRUE, TRUE, NOW(), NOW()),
  ('nightly', '每夜构建', '最新提交构建，不保证稳定', 'your-org/neko-status', NULL, TRUE, FALSE, NOW(), NOW());
```

### `update_releases` — 版本发布记录（缓存）

> 后端定时同步 GitHub Releases API 的结果，避免前端直接打 GitHub API 导致限流。

```sql
CREATE TABLE update_releases (
  id              BIGSERIAL PRIMARY KEY,
  channel_id      VARCHAR(16) NOT NULL REFERENCES update_channels(id),
  version         VARCHAR(32) NOT NULL,             -- 如 1.1.0
  tag_name        VARCHAR(64) NOT NULL,             -- 如 v1.1.0
  release_date    DATE NOT NULL,
  is_latest       BOOLEAN NOT NULL DEFAULT FALSE,
  is_forced       BOOLEAN NOT NULL DEFAULT FALSE,   -- 是否强制更新
  changelog       TEXT,
  min_version     VARCHAR(32),                      -- 能升级过来的最低版本（用于跨版本升级检查）

  -- Windows 安装包
  win_exe_url     TEXT,
  win_exe_sha256  VARCHAR(64),
  win_exe_size    BIGINT,

  -- zip 包（用于手动安装或增量更新）
  win_zip_url     TEXT,
  win_zip_sha256  VARCHAR(64),
  win_zip_size    BIGINT,

  github_release_id BIGINT,                        -- GitHub Release ID，用于去重
  synced_at       TIMESTAMP NOT NULL DEFAULT NOW(),

  UNIQUE(channel_id, version)
);

CREATE INDEX idx_releases_channel ON update_releases(channel_id, is_latest, release_date DESC);
```

### `device_stream_keys` — 推流密钥管理（v1.1.0 新增）

> 每台设备对应唯一 Stream Key，用于 SRS RTMP 推流鉴权。  
> 关联文档：《SRS直播推流-后端实现方案.md》

```sql
CREATE TABLE device_stream_keys (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  device_id   TEXT    NOT NULL UNIQUE,  -- 关联 devices 表，一设备一 Key
  stream_key  TEXT    NOT NULL UNIQUE,  -- 格式：nk_{deviceId前8位}_{8位随机hex}
  created_at  DATETIME DEFAULT CURRENT_TIMESTAMP,
  reset_at    DATETIME                  -- 最近一次重置时间
);

CREATE INDEX idx_stream_keys_device ON device_stream_keys(device_id);
CREATE INDEX idx_stream_keys_key ON device_stream_keys(stream_key);
```

**安全要求**：

- Stream Key 使用 `crypto.randomBytes()` 生成，严禁 `Math.random()`
- 日志中掩码处理：`nk_dev_abc1_****`，不得出现完整 Key 明文

### `stream_sessions` — 直播会话记录（v1.2.0 预留，可提前建表）

> 记录每次推流的开始/结束时间、时长等信息，由 SRS `on_publish` / `on_unpublish` 回调触发写入。

```sql
CREATE TABLE stream_sessions (
  id            INTEGER PRIMARY KEY AUTOINCREMENT,
  device_id     TEXT    NOT NULL,
  stream_key    TEXT    NOT NULL,
  started_at    DATETIME DEFAULT CURRENT_TIMESTAMP,
  ended_at      DATETIME,
  duration_sec  INTEGER,   -- 结束时计算并写入
  peak_viewers  INTEGER DEFAULT 0
);

CREATE INDEX idx_stream_sessions_device ON stream_sessions(device_id, started_at DESC);
```

## 数据库维护计划

### 定时任务（建议每日执行）

```sql
-- 1. 清理 90 天前的状态记录
DELETE FROM status_reports
WHERE received_at < NOW() - INTERVAL '90 days';

-- 2. 清理 30 天前的更新检查日志
DELETE FROM update_check_logs
WHERE checked_at < NOW() - INTERVAL '30 days';

-- 3. 同步 GitHub Releases 缓存（由应用层调用 GitHub API 实现）
```

### 索引与性能优化

- `status_reports` 数据增长快，建议使用 **PostgreSQL 表分区**（按月）
- `screenshots` 的图片文件存在磁盘/对象存储（OSS/S3），数据库只存元数据
- 给 `devices.last_seen_at` 建索引，便于快速判断设备在线状态

## Prisma Schema（Node.js 参考）

```prisma
// schema.prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model Device {
  id            String    @id @default(cuid())
  deviceName    String    @map("device_name")
  hostname      String?
  platform      String    @default("win32")
  macHash       String?   @map("mac_hash")
  clientVersion String?   @map("client_version")
  createdAt     DateTime  @default(now()) @map("created_at")
  lastSeenAt    DateTime? @map("last_seen_at")
  isActive      Boolean   @default(true) @map("is_active")

  apiKeys       ApiKey[]
  statusReports StatusReport[]
  screenshots   Screenshot[]

  @@map("devices")
}

model ApiKey {
  id         String    @id @default(cuid())
  device     Device    @relation(fields: [deviceId], references: [id])
  deviceId   String    @map("device_id")
  keyHash    String    @unique @map("key_hash")
  keyPrefix  String    @map("key_prefix")
  createdAt  DateTime  @default(now()) @map("created_at")
  expiresAt  DateTime? @map("expires_at")
  revokedAt  DateTime? @map("revoked_at")
  lastUsedAt DateTime? @map("last_used_at")

  @@map("api_keys")
}
```
